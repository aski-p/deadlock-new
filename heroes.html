영웅 API 엔드포인트

GET /api/heroes
모든 영웅 목록을 조회합니다.
GET /api/heroes/{id}
특정 영웅의 상세 정보를 조회합니다.
GET /api/heroes/{id}/abilities
특정 영웅의 스킬 정보를 조회합니다.
// JavaScript 예제
fetch('https://deadlock-api.com/api/heroes')
  .then(response => response.json())
  .then(heroes => {
    console.log('영웅 목록:', heroes);
    heroes.forEach(hero => {
      console.log(`${hero.name} (${hero.role})`);
    });
  });

// 특정 영웅 정보 조회
fetch('https://deadlock-api.com/api/heroes/1')
  .then(response => response.json())
  .then(hero => {
    console.log('영웅 정보:', hero);
    console.log('스킬 목록:', hero.abilities);
  });

사용 예제

// JavaScript 예제
fetch('https://deadlock-api.com/api/heroes')
  .then(response => response.json())
  .then(data => {
    console.log('영웅 데이터:', data);
  });

// 특정 영웅 정보 가져오기
fetch('https://deadlock-api.com/api/heroes/1')
  .then(response => response.json())
  .then(hero => {
    console.log('영웅 이름:', hero.name);
  });
                    
# Python 예제
import requests

# 모든 영웅 조회
response = requests.get('https://deadlock-api.com/api/heroes')
heroes = response.json()
print(f'영웅 수: {len(heroes)}')

# 특정 영웅 조회
hero_response = requests.get('https://deadlock-api.com/api/heroes/1')
hero = hero_response.json()
print(f'영웅 이름: {hero["name"]}')
